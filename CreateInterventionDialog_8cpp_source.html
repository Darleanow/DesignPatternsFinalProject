111/**
112 * @brief Connects signals and slots for UI interactivity and validation.
113 */
114void CreateInterventionDialog::setup_connects()
115{
116 connect(m_buttons, &QDialogButtonBox::accepted, this, &QDialog::accept);
117 connect(m_buttons, &QDialogButtonBox::rejected, this, &QDialog::reject);
118
119 auto setup_combo = [this](QComboBox *combo, auto handler) {
120 combo->setEditable(false);
121 connect(
122 combo, QOverload<int>::of(&QComboBox::activated), this,
123 [combo, handler](int) {
124 combo->hidePopup();
125 if (auto *view = combo->view())
126 view->window()->close();
127 QTimer::singleShot(0, handler);
128 }
129 );
130 };
131
132 setup_combo(m_expertise_input, [this] { update_ok_state(); });
133 setup_combo(m_complexity_input, [this] { update_ok_state(); });
134 setup_combo(m_type_input, [this] { update_ok_state(); });
135 setup_combo(m_technician_input, [this] { update_ok_state(); });
136
137 connect(m_date_input, &QDateEdit::dateChanged, this, &CreateInterventionDialog::update_ok_state);
138 connect(m_duration_input, &QSpinBox::valueChanged, this, &CreateInterventionDialog::update_ok_state);
139 connect(m_location_input, &QLineEdit::textChanged, this, &CreateInterventionDialog::update_ok_state);
140}

162void CreateInterventionDialog::update_ok_state()
163{

165 m_duration_input->value() > 0 &&
166 m_expertise_input->currentIndex() > 0 &&
167 m_complexity_input->currentIndex() > 0 &&
168 m_type_input->currentIndex() > 0 &&
169 !m_location_input->text().isEmpty();

171 m_buttons->button(QDialogButtonBox::Ok)->setEnabled(ok);

185/**
186 * @brief Collects all data from the form into an InterventionData structure.
187 *
188 * @return A filled InterventionData instance if the form is valid, otherwise std::nullopt.
189 */
190std::optional<InterventionData> CreateInterventionDialog::intervention_data() const
191{
192 if (!m_buttons->button(QDialogButtonBox::Ok)->isEnabled())
193 return std::nullopt;
194
195 InterventionData data;
196 auto d = m_date_input->date();
197 data.date = std::chrono::year{d.year()} / unsigned(d.month()) / unsigned(d.day());
198 data.anticipated_duration = std::chrono::hours{m_duration_input->value()};
199 data.expertise_required = static_cast<ExpertiseField>(m_expertise_input->currentData().toInt());
200 data.complexity = static_cast<InterventionComplexity>(m_complexity_input->currentIndex() - 1);
201 data.type = static_cast<InterventionType>(m_type_input->currentData().toInt());
202 data.status = InterventionStatus::PENDING;
203 data.location = Location{
204 m_location_input->text().toStdString(),
205 {0, 0} // Default GPS coords
206 };
207 data.notes = m_notes_input->text().toStdString();
208 QVariant v = m_technician_input->currentData();
209 data.technician_id = v.isValid() ? std::optional{v.toInt()} : std::nullopt;
210
211 return data;
212}

214/**
215 * @brief Tints an icon with a given color.

217 * @param path File path to the icon.
218 * @param color The tint color to apply.

220 */
221QIcon CreateInterventionDialog::tinted_icon(const QString &path, const QColor &color) const
222{
223 QPixmap src(path);
224 QPixmap dst(src.size());
225 dst.fill(Qt::transparent);
226 QPainter p(&dst);
227 p.setCompositionMode(QPainter::CompositionMode_Source);
228 p.drawPixmap(0, 0, src);
229 p.setCompositionMode(QPainter::CompositionMode_SourceIn);
230 p.fillRect(dst.rect(), color);
231 p.end();
232 return QIcon(dst);
233}
