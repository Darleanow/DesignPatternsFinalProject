28void CreateTechnicianDialog::setup_ui()
29{
30 setWindowTitle("New Technician");
31
32 auto *form = new QFormLayout(this);
33
34 m_name_line_edit = new QLineEdit(this);
35 form->addRow("Full Name", m_name_line_edit);
36
37 m_expertise_list = new QListWidget(this);
38 m_expertise_list->setSelectionMode(QAbstractItemView::MultiSelection);
39
40 static const std::vector<ExpertiseField> all_fields = {
41 ExpertiseField::ELECTRICAL, ExpertiseField::NETWORKING,
42 ExpertiseField::HVAC, ExpertiseField::PLUMBING,
43 ExpertiseField::SECURITY, ExpertiseField::SOFTWARE,
44 ExpertiseField::HARDWARE, ExpertiseField::FIRE_SAFETY,
45 ExpertiseField::STRUCTURAL, ExpertiseField::GENERALIST
46 };
47
48 for (const auto &field : all_fields)
49 m_expertise_list->addItem(QString::fromStdString(to_string(field)));
50
51 form->addRow("Expertise Fields", m_expertise_list);
52
53 m_buttons = new QDialogButtonBox(
54 QDialogButtonBox::Ok | QDialogButtonBox::Cancel, this
55 );
56 form->addRow(m_buttons);
57
58 resize(700, 500);
59}
