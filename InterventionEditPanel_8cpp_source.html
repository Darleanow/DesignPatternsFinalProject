109void InterventionEditPanel::setup_connects()
110{
111 auto setup_combo = [this](QComboBox *combo) {
112 combo->setEditable(false);
113 connect(
114 combo, QOverload<int>::of(&QComboBox::activated), combo,
115 [combo](int) {
116 combo->hidePopup();
117 if (auto *v = combo->view())
118 v->window()->close();
119 }
120 );
121 };
122
123 setup_combo(m_expertise_input);
124 setup_combo(m_complexity_input);
125 setup_combo(m_type_input);
126 setup_combo(m_status_input);

130/**
131 * @brief Loads existing data into the form for editing.
132 *
133 * @param d The intervention data to populate the form with.
134 */
135void InterventionEditPanel::load(const InterventionData &d)
136{
137 m_id = d.id;
138 QDate qd(
139 int(d.date.year()), int(static_cast<unsigned>(d.date.month())),
140 int(static_cast<unsigned>(d.date.day()))
141 );
142 m_date_input->setDate(qd);
143 m_duration_input->setValue(int(d.anticipated_duration.count()));
144 m_expertise_input->setCurrentIndex(int(d.expertise_required));
145 m_complexity_input->setCurrentIndex(int(d.complexity));
146 m_type_input->setCurrentIndex(int(d.type));
147 m_status_input->setCurrentIndex(int(d.status));
148 m_location_input->setText(QString::fromStdString(d.location.address));
149 m_notes_input->setText(QString::fromStdString(d.notes));
150
151 m_tech_controller->update_technicians();
152 int idx = 0;
153 if (d.technician_id) {
154 for (int i = 1; i < m_technician_input->count(); ++i) {
155 if (m_technician_input->itemData(i).toInt() == *d.technician_id) {
156 idx = i;
157 break;
158 }
159 }
160 }
161 m_technician_input->setCurrentIndex(idx);
162 setVisible(true);
163}

175/**
176 * @brief Collects and returns the current form data as a new InterventionData structure.
177 *
178 * @return The updated InterventionData.
179 */
180InterventionData InterventionEditPanel::to_data() const
181{
182 InterventionData data;
183 data.id = m_id;
184 auto date = m_date_input->date();
185 data.date = std::chrono::year{date.year()} / unsigned(date.month()) / unsigned(date.day());
186 data.anticipated_duration = std::chrono::hours{m_duration_input->value()};
187 data.expertise_required = static_cast<ExpertiseField>(m_expertise_input->currentIndex());
188 data.complexity = static_cast<InterventionComplexity>(m_complexity_input->currentIndex());
189 data.type = static_cast<InterventionType>(m_type_input->currentIndex());
190 data.status = static_cast<InterventionStatus>(m_status_input->currentIndex());
191 data.location = Location{
192 m_location_input->text().toStdString(),
193 {0, 0}
194 };
195 data.notes = m_notes_input->text().toStdString();
196 QVariant v = m_technician_input->currentData();
197 data.technician_id = v.isValid() ? std::optional{v.toInt()} : std::nullopt;
198 data.completed = false;
199 return data;
200}
