58/**
59 * @brief Connects UI signals and slots to handle creation and editing of interventions.
60 */
61void InterventionPage::setup_connects()
62{
63 connect(m_create_button, &QPushButton::clicked, this, [this]() {
64 try {
65 if (m_create_dialog->exec() == QDialog::Accepted) {
66 auto data = m_create_dialog->intervention_data();
67 if (data.has_value()) {
68 data->id = InterventionRepositoryProxy::instance().next_id();
69 auto intervention =
70 InterventionCreatorFacade::create_intervention(data.value());
71 InterventionRepositoryProxy::instance().save(intervention);
72 }
73 }
74 } catch (const AccessDeniedException &e) {
75 QMessageBox::warning(
76 this, "Access Denied", QString::fromStdString(e.what())
77 );
78 }
79 });
80
81 connect(
82 m_list_view, &InterventionsListView::intervention_selected, this,
83 [this](int id) {
84 m_selected_intervention_id = id;
85 auto opt = InterventionRepositoryProxy::instance().find_by_id(id);
86 if (opt.has_value()) {
87 const auto &intervention = opt.value();
88 m_edit_panel->load(intervention->data());
89 }
90
91 update_ui_for_current_role();
92 }
93 );
94
95 connect(m_edit_panel->save_button(), &QPushButton::clicked, this, [this]() {
96 if (!m_selected_intervention_id.has_value())
97 return;
98
99 try {
100 auto data = m_edit_panel->to_data();
101 data.id = *m_selected_intervention_id;
102 auto updated = InterventionCreatorFacade::create_intervention(data);
103 InterventionRepositoryProxy::instance().save(updated);
104 } catch (const AccessDeniedException &e) {
105 QMessageBox::warning(
106 this, "Access Denied", QString::fromStdString(e.what())
107 );
108 }
109 });
110}

113 * @brief Updates the UI components according to the current user's role.
114 */
115void InterventionPage::update_ui_for_current_role()
116{
117 bool can_edit =
118 AuthenticationService::instance().has_role(UserRole::TECHNICIAN);
119
120 m_create_button->setEnabled(can_edit);
121
122 if (m_edit_panel) {
123 m_edit_panel->save_button()->setEnabled(can_edit);
124
125 if (!can_edit && m_edit_panel->isVisible()) {
126 QLabel *warning =
127 new QLabel("View only mode. Please log in to edit.", m_edit_panel);
128 warning->setStyleSheet("color: red;");
129
130 QFormLayout *layout = qobject_cast<QFormLayout *>(m_edit_panel->layout());
131 if (layout) {
132 layout->addRow("", warning);
133 }
134 }
135 }
