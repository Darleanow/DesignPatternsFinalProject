60/**
61 * @brief Updates the list view with the latest interventions from the repository.
62 *
63 * Each item shows date, expertise, complexity, duration, location, and technician.
64 */
65void InterventionsListView::on_repository_updated()
66{
67 QStringList list;
68 const auto all = InterventionRepository::instance().get_all();
69 const auto &techs = TechnicianRepository::instance().get_all();
70
71 for (auto &iv : all) {
72 const auto &d = iv->data();
73
74 int year = static_cast<int>(d.date.year());
75 int month = static_cast<int>(static_cast<unsigned>(d.date.month()));
76 int day = static_cast<int>(static_cast<unsigned>(d.date.day()));
77 QDate qdate(year, month, day);
78
79 QString expertise = QString::fromStdString(to_string(d.expertise_required));
80 QString complexity = complexity_names.value(int(d.complexity));
81 QString duration = QStringLiteral("%1h").arg(d.anticipated_duration.count());
82 QString location = QString::fromStdString(d.location.address);
83
84 QString tech_name = "Unassigned";
85 if (d.technician_id) {
86 for (auto &t : techs) {
87 if (t->get_id() == *d.technician_id) {
88 tech_name = QString::fromStdString(t->get_name());
89 break;
90 }
91 }
92 }
93
94 QString s = QStringLiteral("[%1] %2 — %3 — %4 — %5 — %6")
95 .arg(qdate.toString("yyyy‑MM‑dd"))
96 .arg(expertise)
97 .arg(complexity)
98 .arg(duration)
99 .arg(location)
100 .arg(tech_name);
101
102 list << s;
103 }
104
105 m_model->setStringList(list);
106}
