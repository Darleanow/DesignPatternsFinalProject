145/**
146 * @brief Attempts to log the user in using the entered credentials.
147 */
148void LoginPage::attempt_login()
149{
150 QString username = m_username_input->text();
151 QString password = m_password_input->text();
152
153 if (AuthenticationService::instance().login(username.toStdString(), hash_password(password.toStdString()))) {
154 auto current_user = AuthenticationService::instance().current_user();
155 update_ui_for_logged_in_user(current_user);
156 emit login_successful(current_user->role());
157 clear_fields();
158 } else {
159 m_status_label->setText("Invalid username or password");
160 QTimer::singleShot(3000, m_status_label, &QLabel::clear);
161 }
162}

184/**
185 * @brief Updates the UI to show the logged-in state.
186 *
187 * @param user The currently logged-in user.
188 */
189void LoginPage::update_ui_for_logged_in_user(std::shared_ptr<User> user)
190{
191 if (!user)
192 return;
193
194 m_username_input->setVisible(false);
195 m_password_input->setVisible(false);
196 m_login_button->setVisible(false);
197 m_guest_button->setVisible(false);
198
199 QString role_name;
200 switch (user->role()) {
201 case UserRole::ADMIN: role_name = "Admin"; break;
202 case UserRole::TECHNICIAN: role_name = "Technician"; break;
203 case UserRole::GUEST: role_name = "Guest"; break;
204 }
205
206 m_user_info->setText(QString("Logged in as: %1 (%2)")
207 .arg(QString::fromStdString(user->username()))
208 .arg(role_name));
209 m_user_info->setVisible(true);
210 m_logout_button->setVisible(true);
211}

254/**
255 * @brief Returns a tinted version of an icon loaded from a given file path.
256 *
257 * @param path File path to the icon.
258 * @param color Tint color.
259 * @return A QIcon with the applied tint.
260 */
261QIcon LoginPage::tinted_icon(const QString &path, const QColor &color) const
262{
263 QPixmap src(path);
264 if (src.isNull()) {
265 return QIcon();
266 }
267
268 QPixmap dst(src.size());
269 dst.fill(Qt::transparent);
270
271 QPainter p(&dst);
272 if (!p.isActive()) {
273 return QIcon(src);
274 }
275
276 p.setCompositionMode(QPainter::CompositionMode_Source);
277 p.drawPixmap(0, 0, src);
278 p.setCompositionMode(QPainter::CompositionMode_SourceIn);
279 p.fillRect(dst.rect(), color);
280 p.end();
281
282 return QIcon(dst);
283}
