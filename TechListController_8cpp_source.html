92 bool has_expertise = m_expertise_input->currentData().isValid();

97 if (!has_expertise || !has_complexity) {
98 m_technician_input->setEnabled(false);

100 }
101
102 auto required =
103 static_cast<ExpertiseField>(m_expertise_input->currentData().toInt());
104 int cidx = placeholder ? m_complexity_input->currentIndex() - 1

106 auto comp = static_cast<InterventionComplexity>(cidx);
107 bool allow_gen = comp <= InterventionComplexity::MEDIUM;
108
109 for (auto tech : TechnicianRepository::instance().get_all()) {
110 if (tech->has_expertise(required) ||
111 (allow_gen && tech->has_expertise(ExpertiseField::GENERALIST))) {
112 m_technician_input->addItem(
113 QString::fromStdString(tech->get_name()), tech->get_id()
114 );
115 }
116 }
117
118 int to_select = 0;
119 if (prev_id >= 0) {
120 for (int i = 1; i < m_technician_input->count(); ++i) {
121 if (m_technician_input->itemData(i).toInt() == prev_id) {
122 to_select = i;
123 break;
124 }
125 }
126 }
127 m_technician_input->setCurrentIndex(to_select);
128 m_technician_input->setEnabled(true);
