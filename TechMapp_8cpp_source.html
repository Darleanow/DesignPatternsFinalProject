57/**
58 * @brief Loads the RobotoMono font and applies it to the application.
59 *
60 * @param app Reference to the QApplication instance.
61 */
62void TechMapp::setup_font(QApplication &app)
63{
64 int id = QFontDatabase::addApplicationFont(":/styles/fonts/RobotoMono.ttf");
65
66 if (id == -1) {
67 qWarning("Failed to load RobotoMono font.");
68 return;
69 }
70
71 QStringList families = QFontDatabase::applicationFontFamilies(id);
72 if (families.isEmpty()) {
73 qWarning("Font loaded, but no families found.");
74 return;
75 }
76
77 QFont font(families.at(0));
78 font.setPointSize(12);
79 font.setHintingPreference(QFont::PreferFullHinting);
80 font.setStyleStrategy(QFont::PreferAntialias);
81 app.setFont(font);
82}

155/**
156 * @brief Updates the UI elements and role label depending on the user role.
157 *
158 * @param role The current authenticated user role.
159 */
160void TechMapp::update_ui_for_role(UserRole role)
161{
162 QString role_name;
163 switch (role) {
164 case UserRole::ADMIN:
165 role_name = "Admin";
166 break;
167 case UserRole::TECHNICIAN:
168 role_name = "Technician";
169 break;
170 case UserRole::GUEST:
171 default:
172 role_name = "Guest";
173 break;
174 }
175
176 auto user = AuthenticationService::instance().current_user();
177 QString username = user ? QString::fromStdString(user->username()) : "";
178
179 if (!username.isEmpty()) {
180 m_user_status_label->setText(QString("%1 (%2)").arg(username).arg(role_name));
181 } else {
182 m_user_status_label->setText(role_name);
183 }
184}
