27void TechnicianEditPanel::setup_ui()
28{
29 auto *layout = new QFormLayout(this);
30
31 m_name_input = new QLineEdit(this);
32 layout->addRow("Full Name", m_name_input);
33
34 m_expertise_list = new QListWidget(this);
35 m_expertise_list->setSelectionMode(QAbstractItemView::MultiSelection);
36
37 static const std::vector<ExpertiseField> all_fields = {
38 ExpertiseField::ELECTRICAL, ExpertiseField::NETWORKING,
39 ExpertiseField::HVAC, ExpertiseField::PLUMBING,
40 ExpertiseField::SECURITY, ExpertiseField::SOFTWARE,
41 ExpertiseField::HARDWARE, ExpertiseField::FIRE_SAFETY,
42 ExpertiseField::STRUCTURAL, ExpertiseField::GENERALIST
43 };
44
45 for (const auto &field : all_fields)
46 m_expertise_list->addItem(QString::fromStdString(to_string(field)));
47
48 layout->addRow("Expertise Fields", m_expertise_list);
49
50 m_save_button = new QPushButton("Save Changes", this);
51 layout->addRow("", m_save_button);
52
53 layout->setVerticalSpacing(12);
54 layout->setContentsMargins(8, 8, 8, 8);
55
56 setVisible(false);
57}

84void TechnicianEditPanel::set_expertises(const std::vector<ExpertiseField> &expertises)
85{
86 for (int i = 0; i < m_expertise_list->count(); ++i) {
87 auto *item = m_expertise_list->item(i);
88 item->setSelected(false);
89 for (const auto &field : expertises) {
90 if (item->text() == QString::fromStdString(to_string(field)))
91 item->setSelected(true);
92 }
93 }
94}

132 else if (text == "Security")
133 result.push_back(ExpertiseField::SECURITY);
134 else if (text == "Software")
135 result.push_back(ExpertiseField::SOFTWARE);
136 else if (text == "Hardware")
137 result.push_back(ExpertiseField::HARDWARE);
138 else if (text == "Fire Safety")
139 result.push_back(ExpertiseField::FIRE_SAFETY);
