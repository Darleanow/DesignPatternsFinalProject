58/**
59 * @brief Connects user actions (buttons, selection) to their respective logic.
60 */
61void TechniciansPage::setup_connects()
62{
63 connect(m_create_button, &QPushButton::clicked, m_create_dialog, [this]() {
64 try {
65 if (m_create_dialog->exec() == QDialog::Accepted) {
66 auto full_name = m_create_dialog->full_name();
67 auto selected_expertises = m_create_dialog->selected_expertises();
68
69 auto &technician_repository = TechnicianRepositoryProxy::instance();
70
71 std::shared_ptr<ATechnician> technician =
72 std::make_shared<BasicTechnician>(
73 technician_repository.next_id(), full_name.toStdString()
74 );
75
76 for (const auto &expertise : selected_expertises) {
77 technician = std::make_shared<ExpertiseDecorator>(technician, expertise);
78 }
79
80 technician_repository.save(technician);
81 }
82 } catch (const AccessDeniedException &e) {
83 QMessageBox::warning(
84 this, "Access Denied", QString::fromStdString(e.what())
85 );
86 }
87 });
88
89 connect(
90 m_list_view, &TechniciansListView::technician_selected, this,
91 [this](int id) {
92 m_selected_technician_id = id;
93
94 const auto tech = TechnicianRepositoryProxy::instance().find_by_id(id);
95 if (!tech)
96 return;
97
98 m_edit_panel->load_technician(
99 tech->get()->get_id(),
100 QString::fromStdString(tech->get()->get_name()),
101 tech->get()->get_expertise()
102 );
103
104 m_edit_panel->setVisible(true);
105
106 update_ui_for_current_role();
107 }
108 );
109
110 connect(m_edit_panel->save_button(), &QPushButton::clicked, this, [this]() {
111 if (!m_selected_technician_id.has_value())
112 return;
113
114 try {
115 const auto id = *m_selected_technician_id;
116 const auto name = m_edit_panel->name().toStdString();
117 const auto expertises = m_edit_panel->selected_expertises();
118
119 std::shared_ptr<ATechnician> technician =
120 std::make_shared<BasicTechnician>(id, name);
121
122 for (const auto &exp : expertises)
123 technician = std::make_shared<ExpertiseDecorator>(technician, exp);
124
125 TechnicianRepositoryProxy::instance().save(technician);
126 } catch (const AccessDeniedException &e) {
127 QMessageBox::warning(
128 this, "Access Denied", QString::fromStdString(e.what())
129 );
130 }
131 });
132}

134/**
135 * @brief Updates UI elements according to the user's role.
136 */
137void TechniciansPage::update_ui_for_current_role()
138{
139 bool is_admin = AuthenticationService::instance().has_role(UserRole::ADMIN);
140
141 m_create_button->setEnabled(is_admin);
142 m_create_button->setVisible(is_admin);
143
144 if (m_edit_panel) {
145 m_edit_panel->save_button()->setEnabled(is_admin);
146
147 if (!is_admin && m_edit_panel->isVisible()) {
148 QLabel *warning = new QLabel(
149 "View only mode. Admin access required to edit.", m_edit_panel
150 );
151 warning->setStyleSheet("color: red;");
152
153 QFormLayout *layout = qobject_cast<QFormLayout *>(m_edit_panel->layout());
154 if (layout) {
155 layout->addRow("", warning);
156 }
157 }
158 }
159}
